# linux-diversos

# iniciar noip
- [sudo]  /usr/local/bin/noip2

# link simbolico (alias)
- ln -s /var/log/boot.log /home/usuario/novoarquivo -> cria um referencia dentro /home/usuario/novarquivo apontando para o boot.log
- ls -lah (mostra o link)

# liberar firewall
- install ufw
- ufw disable / enable
- ufw allow 3389/tcp
